Introducción

En este capítulo se presentan casos prácticos implementables con Apache Spark Structured Streaming utilizando Apache Kafka como fuente de eventos y Parquet como almacenamiento de salida (data lake).

Lectura y parseo desde Kafka

from pyspark.sql import SparkSession
from pyspark.sql.functions import col, from_json
from pyspark.sql.types import StructType, StructField, StringType, TimestampType, DoubleType

spark = SparkSession.builder.appName("TFG-Streaming-Kafka-Parquet").getOrCreate()

bootstrap_servers = "kafka:9092"

raw = (spark.readStream
       .format("kafka")
       .option("kafka.bootstrap.servers", bootstrap_servers)
       .option("subscribe", "mi_topic")
       .option("startingOffsets", "latest")
       .load())

schema = StructType([
    StructField("ts", TimestampType(), True),
    StructField("id", StringType(), True),
    StructField("value", DoubleType(), True),
    StructField("type", StringType(), True)
])

events = (raw.select(from_json(col("value").cast("string"), schema).alias("e"))
          .select("e.*"))

Caso práctico 1: Clickstream (usuarios activos y top URLs)

from pyspark.sql.functions import window, approx_count_distinct

active_users_1m = (events
  .withWatermark("ts", "10 minutes")
  .groupBy(window(col("ts"), "1 minute"))
  .agg(approx_count_distinct("userId").alias("active_users")))

urls_5m = (events
  .withWatermark("ts", "10 minutes")
  .groupBy(window(col("ts"), "5 minutes"), col("url"))
  .count())

q1 = (active_users_1m.writeStream
      .outputMode("append")
      .format("parquet")
      .option("path", "/data/out/case1_active_users_1m")
      .option("checkpointLocation", "/data/chk/case1_active_users_1m")
      .start())

q2 = (urls_5m.writeStream
      .outputMode("append")
      .format("parquet")
      .option("path", "/data/out/case1_top_urls_5m")
      .option("checkpointLocation", "/data/chk/case1_top_urls_5m")
      .start())

spark.streams.awaitAnyTermination()

Caso práctico 2: Detección de fraude (picos de transacciones)

from pyspark.sql.functions import lit

N = 10
alerts = (tx
  .withWatermark("ts", "15 minutes")
  .groupBy(window(col("ts"), "2 minutes", "30 seconds"), col("cardId"))
  .count()
  .filter(col("count") >= lit(N))
  .withColumn("rule", lit("BURST_TX_COUNT"))
  .withColumn("threshold", lit(N)))

q = (alerts.writeStream
     .outputMode("append")
     .format("parquet")
     .option("path", "/data/out/case2_fraud_alerts")
     .option("checkpointLocation", "/data/chk/case2_fraud_alerts")
     .start())

q.awaitTermination()

Caso práctico 3: ETL continuo Bronze → Silver con foreachBatch

def process_batch(batch_df, batch_id: int):
    clean = (batch_df
             .dropna(subset=["ts", "id", "source", "value"])
             .dropDuplicates(["id"])
             .withColumn("date", col("ts").cast("date")))

    (clean.write
     .mode("append")
     .partitionBy("date")
     .parquet("/data/out/case3_silver_metrics"))

q = (events.writeStream
     .foreachBatch(process_batch)
     .option("checkpointLocation", "/data/chk/case3_bronze_to_silver")
     .start())

q.awaitTermination()

CONCLUSIÓN

Los casos prácticos muestran patrones recurrentes: uso de watermarks para datos tardíos, particionado adecuado para paralelismo y checkpoints para tolerancia a fallos. Documentar y medir (latencia, throughput) es clave en un TFG.
